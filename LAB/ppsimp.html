Core Definitions (3 Marks Each)

1. Array: Definition, Properties & Syntax

Definition: An array is a fixed-size, linear, and homogeneous collection of data elements stored in contiguous memory locations under a single identifier.

Key Properties: (1) Elements share the same data type. (2) Memory is allocated statically at compile-time. (3) Subscript/Index starts from 0 to (size-1).

Syntax: data_type array_name[size];

Example: int marks[5]; reserves 20 bytes (if int is 4 bytes) in a continuous block.

2. 1D Array: Concept & Initialization

Concept: A One-Dimensional array is a list of elements where each item is identified by a single index. It is essentially a mathematical vector in programming.

Logic: Accessing elements takes O(1) time because the compiler calculates address as: BaseAddress + (index * size_of_type).

Initialization: (1) Static: int a[3] = {10, 20, 30}; (2) Dynamic: Using a for loop with scanf.

3. Function: Purpose & Components

Definition: A function is a self-contained module or block of code designed to perform a specific task, promoting the "Divide and Conquer" strategy.

Purpose: (1) Code Reusability (write once, use many). (2) Better Modularity and debugging. (3) Reduction in program size.

Components: Prototype (declaration), Call (execution), and Definition (actual logic body).

4. Sorting: Logic & Basic Techniques

Definition: Sorting is the systematic arrangement of data elements in a specific logical order, typically ascending or descending.

Basic Techniques: (1) Bubble Sort: Repeatedly swaps adjacent elements. (2) Selection Sort: Selects the minimum and places it at the start. (3) Insertion Sort: Inserts elements into a sorted sub-part.

Complexity: Most basic sorts have a worst-case time complexity of O(nÂ²).

5. Parameter Passing: Value vs Reference

Call by Value

Passes a copy of the actual parameter. Changes inside the function do not affect the original variables in main().

Call by Reference

Passes the address (pointer) of the variable. Changes inside the function directly modify the original memory location.

25 Important Programming Solutions

1. Largest Element in Array

High Weightage

Initializes max with first element and updates if current index is greater.

#include <stdio.h>
int main() {
    int a[50], n, i, max;
    printf("Enter size: "); scanf("%d", &n);
    printf("Enter elements: ");
    for(i=0; i<n; i++) scanf("%d", &a[i]);

    max = a[0];
    for(i=1; i<n; i++) if(a[i] > max) max = a[i];

    printf("Largest = %d", max);
    return 0;
}

2. Sum & Average of Array

Adds every element to a running total 'sum' and then divides by n.

#include <stdio.h>
int main() {
    int a[100], n, i; float s = 0;
    scanf("%d", &n);
    for(i=0; i<n; i++) { scanf("%d", &a[i]); s += a[i]; }
    printf("Sum: %.2f, Avg: %.2f", s, s/n);
    return 0;
}

3. Reverse 1D Array

Swaps elements from indices i and (n-1-i) until the middle is reached.

#include <stdio.h>
int main() {
    int a[]={1,2,3,4,5}, n=5, t, i;
    for(i=0; i<n/2; i++) { t=a[i]; a[i]=a[n-1-i]; a[n-1-i]=t; }
    for(i=0; i<n; i++) printf("%d ", a[i]);
    return 0;
}

4. Linear Search Implementation

Iterates index-by-index to match the target key; stops on first find.

#include <stdio.h>
int main() {
    int a[]={10,25,30,45}, k=30, i, f=0;
    for(i=0; i<4; i++) if(a[i]==k) { printf("Pos: %d", i); f=1; break; }
    if(!f) printf("Not Found");
    return 0;
}

5. Binary Search Implementation

Requires sorted array; cuts search range in half every comparison step.

#include <stdio.h>
int main() {
    int a[]={10,20,30,40,50}, l=0, h=4, m, k=40;
    while(l <= h) {
        m = (l+h)/2;
        if(a[m]==k) { printf("Index: %d", m); return 0; }
        if(a[m] < k) l=m+1; else h=m-1;
    }
    printf("Not Found");
    return 0;
}

6. Bubble Sort Logic

Nested loops compare and swap neighbors to push largest value to end.

#include <stdio.h>
int main() {
    int a[]={5,1,4,2}, n=4, i, j, t;
    for(i=0; i<n-1; i++)
        for(j=0; j<n-i-1; j++)
            if(a[j]>a[j+1]) { t=a[j]; a[j]=a[j+1]; a[j+1]=t; }
    for(i=0; i<n; i++) printf("%d ", a[i]);
    return 0;
}

7. Selection Sort Logic

Finds the index of minimum value in unsorted part and swaps with current start.

#include <stdio.h>
int main() {
    int a[]={64,25,12,22}, n=4, i, j, m, t;
    for(i=0; i<n-1; i++) {
        m=i; for(j=i+1; j<n; j++) if(a[j]<a[m]) m=j;
        t=a[m]; a[m]=a[i]; a[i]=t;
    }
    for(i=0; i<n; i++) printf("%d ", a[i]);
    return 0;
}

8. Insertion Sort Logic

Iteratively places current element into its correct slot in a sorted sub-list.

#include <stdio.h>
int main() {
    int a[]={12,11,13,5}, n=4, i, j, k;
    for(i=1; i<n; i++) {
        k=a[i]; j=i-1;
        while(j>=0 && a[j]>k) { a[j+1]=a[j]; j--; }
        a[j+1]=k;
    }
    for(i=0; i<n; i++) printf("%d ", a[i]);
    return 0;
}

9. Swap using Pointers (Call by Reference)

Uses addresses to modify the original variables inside main().

#include <stdio.h>
void swap(int *x, int *y) { int t=*x; *x=*y; *y=t; }
int main() {
    int a=10, b=20;
    swap(&a, &b);
    printf("a:%d b:%d", a, b);
    return 0;
}

10. Factorial Calculation (Function)

Returns the product of all integers from 1 up to n.

#include <stdio.h>
long fact(int n) {
    long f=1; for(int i=1; i<=n; i++) f*=i;
    return f;
}
int main() { printf("%ld", fact(5)); return 0; }

11. Count Even/Odd in Array

Increments counters based on result of modulus 2 operation.

#include <stdio.h>
int main() {
    int a[]={1,2,3,4,5,6}, e=0, o=0;
    for(int i=0; i<6; i++) (a[i]%2==0) ? e++ : o++;

12. Power x^y (Function)

Iteratively multiplies the base x by itself y times.

#include <stdio.h>
double pwr(double x, int y) {
    double r=1; while(y--) r*=x; return r;
}
int main() { printf("%.2f", pwr(2, 3)); return 0; }

13. Second Largest Element

Tracks maximum and updates second largest if a new max is found.

#include <stdio.h>
int main() {
    int a[]={1, 9, 4, 8}, m1=-1, m2=-1;
    for(int i=0; i<4; i++) {
        if(a[i]>m1) { m2=m1; m1=a[i]; }
        else if(a[i]>m2 && a[i]!=m1) m2=a[i];
    }
    printf("2nd Max: %d", m2); return 0;
}

16. Prime Check (Function)

Returns 1 if no divisors are found up to n/2; else returns 0.

#include <stdio.h>
int isP(int n) {
    if(n<2) return 0;
    for(int i=2; i<=n/2; i++) if(n%i==0) return 0;
    return 1;
}
int main() { if(isP(7)) printf("Prime"); return 0; }

17. Merge two Arrays

Sequentially copies first array and then appends second array elements.

#include <stdio.h>
int main() {
    int a[]={1,2}, b[]={3,4}, c[4], i;
    for(i=0; i<2; i++) c[i]=a[i];
    for(i=0; i<2; i++) c[i+2]=b[i];
    for(i=0; i<4; i++) printf("%d ", c[i]); return 0;
}

18. Sum of Prime Elements in Array

Traverses array and adds element to sum only if isPrime logic holds.

#include <stdio.h>
int main() {
    int a[]={2,3,4,5}, s=0, i, j, p;
    for(i=0; i<4; i++) {
        p=1; if(a[i]<2) p=0;
        for(j=2; j<=a[i]/2; j++) if(a[i]%j==0) p=0;
        if(p) s+=a[i];
    }
    printf("Sum: %d", s); return 0;
}

Shifts all subsequent elements one position left to overwrite index.

20. Insert at Specific Position

Shifts elements right from the end to create a gap at the target index.

#include <stdio.h>
int main() {
    int a[10]={1,2,4,5}, n=4, pos=2, val=3, i;
    for(i=n; i>pos; i--) a[i]=a[i-1];
    a[pos]=val;
    for(i=0; i<=n; i++) printf("%d ", a[i]); return 0;
}

21. Max of 3 Numbers (Function)

Uses nested conditional statements to find the largest parameter.

#include <stdio.h>
int mx(int a, int b, int c) {
    if(a>b && a>c) return a;
    return (b>c) ? b : c;
}
int main() { printf("Max: %d", mx(5, 12, 8)); return 0; }

22. Fibonacci Sequence (Function)

Calculates next term by summing the previous two terms in a loop.

#include <stdio.h>
void fib(int n) {
    int a=0, b=1, c;
    for(int i=0; i<n; i++) { printf("%d ", a); c=a+b; a=b; b=c; }
}
int main() { fib(5); return 0; }

23. Check if Array is Sorted

Returns 'Not Sorted' if any element is found to be greater than the next one.

#include <stdio.h>
int main() {
    int a[]={1, 2, 8, 4}, s=1;
    for(int i=0; i<3; i++) if(a[i]>a[i+1]) s=0;
    (s) ? printf("Sorted") : printf("Unsorted"); return 0;
}
